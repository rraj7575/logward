export const TemplateLayout = (args) => {

  const GetWrapper = ({type, children}) => {
    switch (type) {
      case "vertical":
        return <Inline gutter={25}>{children}</Inline>
      case "horizontal": 
        return <Stack gutter={25}>{children}</Stack>
      default:
        return <>{children}</>
    }
  }

  return (
    <>
      <GetWrapper type={args.type}>
        <Tabs type={args.type}>
        <TabList>
          <Tab>Tab One</Tab>
          <Tab>Tab Two</Tab>
          <Tab>Tab Three</Tab>
        </TabList>
        <TabPanels>
          <TabPanel>Content One</TabPanel>
          <TabPanel>Content Two</TabPanel>
          <TabPanel>Content Three</TabPanel>
        </TabPanels>
      </Tabs>
      </GetWrapper>
    </>

);
};
