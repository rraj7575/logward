import { useState } from 'react';

import { Meta, Canvas, Story, ArgsTable } from '@storybook/addon-docs';

import { Tab } from './Tab';

import { Button } from '../Button';

import { InlineCluster } from '@bedrock-layout/inline-cluster';

<Meta title="Base/Tab" component={Tab} />

export const Template = (args) => <Tab {...args} />;

export const TemplateTriggers = (args) => {
  const [tabState, setTabState] = useState(1);
  const tabList = [
    {
      label: 'Tab one',
      value: 1,
    },
    {
      label: 'Tab two',
      value: 2,
    },
    {
      label: 'Tab three',
      value: 3,
    },
  ];
  return (
    <InlineCluster gutter={0}>
      {tabList.map((tab) => (
        <Tab
          key={tab.value}
          active={tab.value === tabState}
          onClick={() => {
            setTabState(tab.value);
          }}
        >
          {tab.label}
        </Tab>
      ))}
    </InlineCluster>
  );
};
