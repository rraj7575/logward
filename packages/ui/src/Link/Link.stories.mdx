import { ArgsTable, Canvas, Meta, Story } from '@storybook/addon-docs';
import { Stack } from '@bedrock-layout/stack';

import { MdArrowBack, MdArrowForward } from 'react-icons/md';

import { Link } from 'ui';

<Meta title="Base/Link" component={Link} />

export const TemplatePrimary = (args) => {
  return (
    <Link to="/signin" {...args}>
      Primary link
    </Link>
  );
};

export const TemplatePrimaryLeading = (args) => {
  return (
    <Link to="/signin" iconLeading={<MdArrowBack />} {...args}>
      Link with Leading icon
    </Link>
  );
};

export const TemplatePrimaryTrailing = (args) => {
  return (
    <Link to="/signin" iconTrailing={<MdArrowForward />} {...args}>
      Link with Trailing icon
    </Link>
  );
};

export const TemplateSecondary = (args) => {
  return (
    <Link appearance="secondary" to="/signin" {...args}>
      Secondary link
    </Link>
  );
};

export const TemplateSecondaryLeading = (args) => {
  return (
    <Link
      appearance="secondary"
      to="/signin"
      iconLeading={<MdArrowBack />}
      {...args}
    >
      Link with Leading icon
    </Link>
  );
};

export const TemplateSecondaryTrailing = (args) => {
  return (
    <Link
      appearance="secondary"
      to="/signin"
      iconTrailing={<MdArrowForward />}
      {...args}
    >
      Link with Trailing icon
    </Link>
  );
};

export const TemplateAssorted = (args) => {
  return (
    <Stack gutter={10}>
      <Link to="/signin" iconLeading={<MdArrowBack />} {...args}>
        Primary link
      </Link>
      <Link to="/signin" iconLeading={<MdArrowBack />} bold={false} {...args}>
        Primary link not bold
      </Link>
      <Link to="/signin" {...args}>
        Primary link without icon
      </Link>
      <Link to="/signin" appearance="secondary" {...args}>
        Secondary link
      </Link>
      <Link
        to="/signin"
        iconLeading={<MdArrowBack />}
        appearance="secondary"
        {...args}
      >
        Secondary link with icon
      </Link>
      <Link to="/signin" appearance="secondary" bold={false} {...args}>
        Secondary link not bold
      </Link>
    </Stack>
  );
};
